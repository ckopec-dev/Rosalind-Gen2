# Rosalind Problem: Error Correction in Reads (Lua Solution)

## Problem Understanding

In this problem, we need to identify and correct sequencing errors in DNA reads. The approach is to:
1. Find reads that appear only once (potential errors)
2. Find reads that appear multiple times (correct reads)
3. For each single-occurrence read, check if it's a Hamming distance of 1 from any correct read
4. If so, correct the error by replacing it with the correct read

## Solution

```lua
function reverse_complement(seq)
    local complement = {
        ['A'] = 'T', ['T'] = 'A', ['G'] = 'C', ['C'] = 'G'
    }
    local result = ""
    for i = #seq, 1, -1 do
        result = result .. complement[seq:sub(i, i)]
    end
    return result
end

function hamming_distance(s1, s2)
    local distance = 0
    for i = 1, #s1 do
        if s1:sub(i, i) ~= s2:sub(i, i) then
            distance = distance + 1
        end
    end
    return distance
end

function solve_error_correction(reads)
    -- Count occurrences of each read
    local count = {}
    for _, read in ipairs(reads) do
        count[read] = (count[read] or 0) + 1
    end
    
    -- Find correct reads (appearing 2+ times) and error reads (appearing 1 time)
    local correct_reads = {}
    local error_reads = {}
    
    for read, freq in pairs(count) do
        if freq >= 2 then
            table.insert(correct_reads, read)
        else
            table.insert(error_reads, read)
        end
    end
    
    -- For each error read, check if it's a Hamming distance of 1 from any correct read
    local corrections = {}
    for _, error_read in ipairs(error_reads) do
        local corrected = false
        for _, correct_read in ipairs(correct_reads) do
            if hamming_distance(error_read, correct_read) == 1 then
                table.insert(corrections, error_read .. "->" .. correct_read)
                corrected = true
                break
            end
        end
        
        -- Also check reverse complement
        if not corrected then
            local rev_comp = reverse_complement(error_read)
            for _, correct_read in ipairs(correct_reads) do
                if hamming_distance(rev_comp, correct_read) == 1 then
                    table.insert(corrections, error_read .. "->" .. correct_read)
                    corrected = true
                    break
                end
            end
        end
    end
    
    return corrections
end
