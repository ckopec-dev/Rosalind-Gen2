# Rosalind Problem: Compute the Score of a Linear Peptide

## Problem Description
Given a linear peptide and a scoring matrix (spectrum), compute the score of the peptide against the spectrum.

## Solution in Perl

```perl
#!/usr/bin/perl
use strict;
use warnings;

# Function to compute the theoretical spectrum of a linear peptide
sub compute_theoretical_spectrum {
    my ($peptide) = @_;
    
    # Amino acid masses
    my %masses = (
        'A' => 71, 'C' => 103, 'D' => 115, 'E' => 129, 'F' => 147,
        'G' => 57, 'H' => 137, 'I' => 113, 'K' => 128, 'L' => 113,
        'M' => 131, 'N' => 114, 'P' => 97, 'Q' => 128, 'R' => 156,
        'S' => 87, 'T' => 101, 'V' => 99, 'W' => 186, 'Y' => 163
    );
    
    my @spectrum = (0);  # Start with 0 (empty peptide)
    
    # Compute all prefix masses
    my $total_mass = 0;
    for my $aa (split //, $peptide) {
        $total_mass += $masses{$aa};
        push @spectrum, $total_mass;
    }
    
    return @spectrum;
}

# Function to compute the score of a peptide against a spectrum
sub compute_score {
    my ($peptide, $spectrum) = @_;
    
    # Get theoretical spectrum
    my @theoretical = compute_theoretical_spectrum($peptide);
    
    # Convert spectrum to hash for faster lookup
    my %spectrum_hash;
    for my $mass (@$spectrum) {
        $spectrum_hash{$mass}++;
    }
    
    # Count matches
    my $score = 0;
    for my $mass (@theoretical) {
        if (exists $spectrum_hash{$mass} && $spectrum_hash{$mass} > 0) {
            $score++;
            $spectrum_hash{$mass}--;
        }
    }
    
    return $score;
}

# Compute and print the score
my $score = compute_score($peptide, \@spectrum);
print "Score: $score\n";
