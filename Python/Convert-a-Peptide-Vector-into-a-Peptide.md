def peptide_vector_to_peptide(vector):
    """
    Convert a peptide vector into a peptide string.
    
    Args:
        vector: List of integers (0 or 1) representing the peptide vector
        
    Returns:
        String representing the peptide
    """
    # Standard amino acid mapping for 3-bit codons
    codon_to_amino = {
        '000': 'G',  # Glycine
        '001': 'A',  # Alanine
        '010': 'S',  # Serine
        '011': 'P',  # Proline
        '100': 'V',  # Valine
        '101': 'T',  # Threonine
        '110': 'C',  # Cysteine
        '111': 'L'   # Leucine
    }
    
    # Convert vector to string
    vector_str = ''.join(map(str, vector))
    
    # Split into 3-bit codons
    peptides = []
    for i in range(0, len(vector_str), 3):
        codon = vector_str[i:i+3]
        if len(codon) == 3:  # Only process complete codons
            amino_acid = codon_to_amino.get(codon, 'X')  # X for unknown
            peptides.append(amino_acid)
    
    return ''.join(peptides)

def peptide_vector_to_peptide_v2(vector):
    """
    Convert a peptide vector into a peptide string (alternative implementation).
    """
    # Mapping from binary string to amino acid
    mapping = {
        '000': 'G', '001': 'A', '010': 'S', '011': 'P',
        '100': 'V', '101': 'T', '110': 'C', '111': 'L'
    }
    
    # Convert vector to string and group into triplets
    vector_str = ''.join(map(str, vector))
    
    # Extract codons and map to amino acids
    result = []
    for i in range(0, len(vector_str), 3):
        if i + 3 <= len(vector_str):
            codon = vector_str[i:i+3]
            result.append(mapping[codon])
    
    return ''.join(result)
